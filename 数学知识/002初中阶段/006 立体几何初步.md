#### 1. 空间直角坐标系（概念 + 左手/右手定则）

**初中生最容易懂的概念**  
把平面坐标系再竖一根 z 轴（向上为正），就变成了空间坐标系。  
任何一个点用三个数 (x, y, z) 就能确定位置。

**左手定则口诀**（中国教材常用）  
- 左手握拳，大拇指朝上（z轴正方向）  
- 食指指向右（x轴正方向）  
- 中指指向前（y轴正方向）

```python
fig = plt.figure(figsize=(12,10))
ax = fig.add_subplot(111, projection='3d')
ax.set_xlim(-2,8); ax.set_ylim(-2,8); ax.set_zlim(-2,8)

# 三条坐标轴
origin = [0,0,0]
ax.quiver(*origin, 6,0,0, color='red',   arrow_length_ratio=0.1)  # x
ax.quiver(*origin, 0,6,0, color='green', arrow_length_ratio=0.1)  # y
ax.quiver(*origin, 0,0,6, color='blue',  arrow_length_ratio=0.1)  # z

ax.text(6.5,0,0,'x轴',color='red',fontsize=16)
ax.text(0,6.5,0,'y轴',color='green',fontsize=16)
ax.text(0,0,6.5,'z轴',color='blue',fontsize=16)

ax.text(3,3,7,'左手定则：大拇指↑z，食指→x，中指→y',fontsize=14,weight='bold')

# 标几个经典空间点
points = np.array([[5,2,3],[1,5,4],[3,1,6],[6,6,1]])
for p in points:
    ax.scatter(*p, color='purple', s=100)
    ax.text(p[0]+0.3,p[1],p[2], f'({p[0]},{p[1]},{p[2]})', fontsize=12)

ax.set_title('空间直角坐标系完整概念图（初中立体几何第一课）',fontsize=20)
plt.show()
```

#### 2. 空间两点距离公式（直接套用勾股定理两次）

**概念讲解**  
先算平面距离，再把 z 轴高度一起开根号！

```python
def dist3d(p1, p2):
    return np.sqrt((p2[0]-p1[0])**2 + (p2[1]-p1[1])**2 + (p2[2]-p1[2])**2)

A = np.array([1,2,3])
B = np.array([4,6,3])
C = np.array([4,6,8])

print(f"AB距离（同高）= {dist3d(A,B):.1f}")   # 5.0
print(f"AC距离 = {dist3d(A,C):.1f}")         # 7.07
print(f"BC距离 = {dist3d(B,C):.1f}")         # 5.0
```

#### 3. 常见多面体（长方体、正方体、棱柱、棱锥）

```python
def draw_cuboid(ax, a=4, b=3, c=2, pos=[0,0,0], color='skyblue', alpha=0.6, label='长方体'):
    x0,y0,z0 = pos
    # 8个顶点
    vertices = np.array([[x0+x, y0+y, z0+z] for x in [0,a] for y in [0,b] for z in [0,c]])
    # 6个面
    faces = [[vertices[j] for j in [0,1,3,2]], [vertices[j] for j in [4,5,7,6]],
             [vertices[j] for j in [0,1,5,4]], [vertices[j] for j in [2,3,7,6]],
             [vertices[j] for j in [0,2,6,4]], [vertices[j] for j in [1,3,7,5]]]
    ax.add_collection3d(Poly3DCollection(faces, facecolors=color, linewidths=1, edgecolors='navy', alpha=alpha))
    ax.text(x0+a/2, y0+b/2, z0+c+1, label, fontsize=14, ha='center')

fig = plt.figure(figsize=(18,10))
ax1 = fig.add_subplot(121, projection='3d')
ax2 = fig.add_subplot(122, projection='3d')

# 长方体
draw_cuboid(ax1, 5,3,4, [1,1,0], 'lightcoral', 0.7, '长方体 长5宽3高4')
ax1.set_title('长方体（三视图的源头）')

# 正方体（特例）
draw_cuboid(ax2, 4,4,4, [2,2,0], 'lightgreen', 0.7, '正方体 边长4')
ax2.set_title('正方体')

for ax in [ax1,ax2]:
    ax.set_xlim(0,10); ax.set_ylim(0,10); ax.set_zlim(0,10)
    ax.set_xlabel('x'); ax.set_ylabel('y'); ax.set_zlabel('z')
plt.show()
```

#### 4. 旋转体（圆柱、圆锥、球）—— 初中最难画的三视图

```python
fig = plt.figure(figsize=(20,6))

# 圆柱
ax1 = fig.add_subplot(131, projection='3d')
theta = np.linspace(0, 2*np.pi, 50)
z = np.linspace(0, 6, 20)
Theta, Z = np.meshgrid(theta, z)
X = 2*np.cos(Theta)
Y = 2*np.sin(Theta)
ax1.plot_surface(X, Y, Z, color='gold', alpha=0.7)
ax1.set_title('圆柱（半径2 高6）')

# 圆锥
ax2 = fig.add_subplot(132, projection='3d')
r = np.linspace(0, 3, 20)
z2 = np.linspace(0, 8, 30)
R, Z2 = np.meshgrid(r, z2)
Theta2 = np.linspace(0, 2*np.pi, 50)
X2 = R * np.cos(Theta2) * (8-Z2)/8
Y2 = R * np.sin(Theta2) * (8-Z2)/8
ax2.plot_surface(X2, Y2, Z2, color='lightblue', alpha=0.8)
ax2.set_title('圆锥（底半径3 高8）')

# 球
ax3 = fig.add_subplot(133, projection='3d')
u = np.linspace(0, 2*np.pi, 50)
v = np.linspace(0, np.pi, 30)
X3 = 3 * np.outer(np.cos(u), np.sin(v))
Y3 = 3 * np.outer(np.sin(u), np.sin(v))
Z3 = 3 * np.outer(np.ones_like(u), np.cos(v))
ax3.plot_surface(X3, Y3, Z3, color='lightpink', alpha=0.8)
ax3.set_title('球（半径3）')

for ax in [ax1,ax2,ax3]:
    ax.set_xlim(-5,5); ax.set_ylim(-5,5); ax.set_zlim(0,10)
    ax.set_xlabel('x'); ax.set_ylabel('y'); ax.set_zlabel('z')
plt.suptitle('旋转体全家福：圆柱、圆锥、球',fontsize=22)
plt.show()
```

#### 5. 表面积与体积公式终极口诀表（中考必背）

| 图形   | 表面积公式（初中必背）                          | 体积公式（初中必背）         |
|--------|------------------------------------------------|------------------------------|
| 长方体 | 2(ab+ac+bc)                                    | abc                          |
| 正方体 | 6a²                                            | a³                           |
| 圆柱   | 2πr(r + h)                                     | πr²h                         |
| 圆锥   | πr(r + √(r²+h²))                               | (1/3)πr²h                    |
| 球     | 4πr²                                           | (4/3)πr³                     |

#### 初中立体几何完整概念总结表

| 知识点         | 初中必背概念                                 | 核心公式                        | 高中/大学对应概念           |
|----------------|----------------------------------------------|---------------------------------|-----------------------------|
| 空间坐标系     | 三个互相垂直的轴                             | (x,y,z)                         | ℝ³ 向量空间                 |
| 空间两点距离   | 平面勾股定理再套一次                         | √[(Δx)²+(Δy)²+(Δz)²]            | 欧氏距离                    |
| 长方体/正方体  | 6个面都是矩形                                | V=abc, S=2(ab+ac+bc)            | 仿射几何                    |
| 圆柱/圆锥      | 旋转得到的体                                 | V=πr²h, V=(1/3)πr²h             | 旋转体积分                  |
| 球             | 到球心距离相等的点                           | V=(4/3)πr³                      | 球几何                      |
| 三视图         | 正、侧、俯视图                               | 实线、虚线规则                  | 投影几何                    |
