#### 1. 一次函数 y = kx + b（初中函数之王）

**初中生最爱记的三句话**  
- k > 0 → 向右上方  
- k < 0 → 向右下方  
- |k| 越大越陡，b 决定截距

```python
x = np.linspace(-6, 6, 400)

# 五条经典一次函数
plt.figure(figsize=(12,8))
plt.axhline(0, color='black', linewidth=1)
plt.axvline(0, color='black', linewidth=1)
plt.grid(True, alpha=0.3)

functions = [
    (2, 3,  'y = 2x + 3',   'red'),
    (1, 0,  'y = x',       'blue'),
    (0.5, -2,'y = 0.5x - 2','green'),
    (-1, 4, 'y = -x + 4',  'purple'),
    (-3, -1,'y = -3x - 1', 'orange')
]

for k, b, label, color in functions:
    y = k * x + b
    plt.plot(x, y, color=color, linewidth=3, label=label)
    # 标出 y 轴截距
    plt.plot(0, b, 'o', color=color, markersize=8)

plt.xlim(-6,6); plt.ylim(-10,10)
plt.xticks(range(-6,7,2)); plt.yticks(range(-10,11,2))
plt.title('一次函数全家福（k 决定方向和陡峭，b 决定截距）', fontsize=18, pad=20)
plt.xlabel('x'); plt.ylabel('y')
plt.legend(fontsize=14, loc='upper left')
plt.show()
```

#### 2. 反比例函数 y = k/x（初中函数最“妖”的一位）

**初中生必背口诀**  
k > 0 → 一三象限；k < 0 → 二四象限；永远不过原点，越靠近原点越“炸”

```python
x1 = np.linspace(0.1, 6, 300)
x2 = np.linspace(-6, -0.1, 300)

plt.figure(figsize=(12,10))
plt.axhline(0, color='black', linewidth=1)
plt.axvline(0, color='black', linewidth=1, linestyle='--', alpha=0.7)
plt.grid(True, alpha=0.3)

ks = [6, 2, -4, -12]
colors = ['red', 'blue', 'green', 'purple']

for k, color in zip(ks, colors):
    y1 = k / x1
    y2 = k / x2
    plt.plot(x1, y1, color=color, linewidth=3)
    plt.plot(x2, y2, color=color, linewidth=3, label=f'y = {k}/x')

# 渐近线标注
plt.text(5, 8, 'x 轴、y 轴是渐近线', fontsize=16, bbox=dict(boxstyle="round", facecolor="wheat"))

plt.xlim(-6,6); plt.ylim(-12,12)
plt.title('反比例函数全家福（双曲线家族）', fontsize=20, pad=20)
plt.xlabel('x (永远不能为 0)'); plt.ylabel('y')
plt.legend(fontsize=14)
plt.show()
```

#### 3. 二次函数 y = ax² + bx + c（初中最美函数）

**初中必背三要素**  
- a > 0 开口向上，最小值  
- a < 0 开口向下，最大值  
- 顶点坐标：(-b/(2a),  (4ac-b²)/(4a) )

```python
x = np.linspace(-6, 6, 400)

plt.figure(figsize=(14,10))
plt.axhline(0, color='black', linewidth=1)
plt.axvline(0, color='black', linewidth=1)
plt.grid(True, alpha=0.3)

cases = [
    (1, -4, 3,   'y = x² - 4x + 3',   'red'),      # 开口向上
    (-1, 2, 5,   'y = -x² + 2x + 5',  'blue'),     # 开口向下
    (0.5, 0, -4, 'y = 0.5x² - 4',      'green'),    # 顶点在 y 轴
    (2, -8, 2,   'y = 2x² - 8x + 2',   'purple'),   # 很陡
]

for a,b,c,label,color in cases:
    y = a*x**2 + b*x + c
    vertex_x = -b/(2*a)
    vertex_y = a*vertex_x**2 + b*vertex_x + c
    plt.plot(x, y, color=color, linewidth=3, label=label)
    plt.plot(vertex_x, vertex_y, 'o', color=color, markersize=10)
    plt.text(vertex_x, vertex_y+1 if a>0 else vertex_y-2, 
             f'顶点({vertex_x:.1f},{vertex_y:.1f})', 
             fontsize=12, ha='center', weight='bold')

plt.xlim(-6,6); plt.ylim(-10,12)
plt.title('二次函数抛物线全家福（初中数学最美的一页）', fontsize=20, pad=20)
plt.xlabel('x'); plt.ylabel('y')
plt.legend(fontsize=14)
plt.show()
```

#### 4. 三兄弟同台大比拼（初中压轴必考）

```python
x_pos = np.linspace(0.1, 6, 300)
x_neg = np.linspace(-6, -0.1, 300)
x_all = np.linspace(-6, 6, 500)

plt.figure(figsize=(14,10))
plt.axhline(0,c='black',lw=1); plt.axvline(0,c='black',lw=1)
plt.grid(True, alpha=0.3)

# 一次函数
plt.plot(x_all, 2*x_all - 1, 'red', linewidth=3, label='一次函数 y = 2x - 1')

# 二次函数
y_quad = 0.5*x_all**2 - 3*x_all + 2
plt.plot(x_all, y_quad, 'blue', linewidth=3, label='二次函数 y = 0.5x² - 3x + 2')

# 反比例函数
plt.plot(x_pos, 6/x_pos, 'green', linewidth=3)
plt.plot(x_neg, 6/x_neg, 'green', linewidth=3, label='反比例函数 y = 6/x')

plt.xlim(-6,6); plt.ylim(-10,10)
plt.title('初中三大函数同台竞技（一次、二次、反比例）', fontsize=22, pad=20)
plt.xlabel('x'); plt.ylabel('y')
plt.legend(fontsize=16, loc='upper right')
plt.show()
```
