```python
# 单位圆 + 任意角现场演示
theta_deg = np.array([0, 30, 45, 60, 90, 120, 135, 150, 180, 210, 225, 240, 270, 300, 315, 330, 360])
theta_rad = np.deg2rad(theta_deg)
x = np.cos(theta_rad)
y = np.sin(theta_rad)

fig, ax = plt.subplots(figsize=(10,10))
circle = plt.Circle((0,0),1,color='gray',fill=False,linewidth=2)
ax.add_patch(circle)
ax.axhline(0,color='k',linewidth=1); ax.axvline(0,color='k',linewidth=1)

# 画射线和点
for i in range(len(theta_deg)):
    if theta_deg[i] in [90,270]: continue
    ax.plot([0,x[i]],[0,y[i]],color='blue',alpha=0.7)
    ax.plot(x[i],y[i],'o',color='red')
    ax.text(x[i]*1.15,y[i]*1.15,f'{theta_deg[i]}°',fontsize=10)

ax.set_xlim(-1.3,1.3); ax.set_ylim(-1.3,1.3); ax.set_aspect('equal')
ax.grid(True,alpha=0.3)
ax.set_title('单位圆 + 任意角三角函数（高中三角第一铁律）',fontsize=20)
plt.show()
```

#### 2. 诱导公式（高考送分公式合集，一共9组）

**终极口诀（倒背如流）**
- 周而复始：360°+α 与 α 完全相同  
- 轴上翻倍：180°+α → 符号取反  
- 负角公式：−α → 正弦取反，余弦同  
- 补角公式：90°−α → 正弦变余弦，余弦变正弦

```python
print("诱导公式终极九连环".center(60,"="))
alpha = 30
print(f"sin(360°+{alpha}°) = sin {alpha}° = {np.sin(np.deg2rad(alpha)):.3f}")
print(f"sin(180°+{alpha}°) = -sin {alpha}° = {np.sin(np.deg2rad(180+alpha)):.3f}")
print(f"sin(180°-{alpha}°) = sin {alpha}° = {np.sin(np.deg2rad(180-alpha)):.3f}")
print(f"cos(90°+{alpha}°) = -sin {alpha}° = {np.cos(np.deg2rad(90+alpha)):.3f}")
print(f"tan(180°-{alpha}°) = -tan {alpha}° = {np.tan(np.deg2rad(180-alpha)):.3f}")
```

#### 3. 和差化积 + 积化和差（高考压轴变形神器）

**四组和差化积公式（必须手推一遍）**
```
sin A ± sin B = 2 sin((A±B)/2) cos((A∓B)/2)
cos A + cos B = 2 cos((A+B)/2) cos((A-B)/2)
cos A - cos B = -2 sin((A+B)/2) sin((A-B)/2)
sin A cos B ± cos A sin B = sin(A±B)
```

```python
# 现场验证和差化积
A, B = 60, 30
print(f"sin{A}° + sin{B}° = {np.sin(np.deg2rad(A))+np.sin(np.deg2rad(B)):.4f}")
print(f"2 sin{(A+B)/2}° cos{(A-B)/2}° = 2*sin{45}°*cos{15}° = {2*np.sin(np.deg2rad(45))*np.cos(np.deg2rad(15)):.4f}")

# 高考最爱：化简 sin(x+10°) + sin(x-10°)
print("\n高考经典：sin(x+10°) + sin(x-10°) = 2 sin x cos 10°")
x = np.linspace(0,360,500)
y1 = np.sin(np.deg2rad(x+10)) + np.sin(np.deg2rad(x-10))
y2 = 2 * np.sin(np.deg2rad(x)) * np.cos(np.deg2rad(10))
print(f"数值完全相等：最大误差 {np.max(np.abs(y1-y2)):.2e}")
```

#### 4. 辅助角公式（高考最后一击）

**终极公式（背会秒杀所有 a sin x + b cos x）**
a sin x + b cos x = √(a²+b²) sin(x + φ)  
其中 tan φ = b/a（注意象限）

```python
# 现场把 3sin x + 4cos x 化成 R sin(x+φ)
a, b = 3, 4
R = np.sqrt(a**2 + b**2)
phi = np.arctan2(b,a)   # 正确的象限

print(f"3sin x + 4cos x = {R} sin(x + {np.rad2deg(phi):.1f}°)")
print(f"最大值 = {R}, 最小值 = {-R}")

# 画图验证
x = np.linspace(0, 2*np.pi, 500)
y1 = 3*np.sin(x) + 4*np.cos(x)
y2 = R * np.sin(x + phi)

plt.figure(figsize=(12,6))
plt.plot(x, y1, 'red', linewidth=3, label='3sin x + 4cos x')
plt.plot(x, y2, 'blue', '--', linewidth=2, label=f'{R:.0f} sin(x + {np.rad2deg(phi):.1f}°)')
plt.title('辅助角公式现场验证（完全重合）')
plt.legend(); plt.grid(alpha=0.3)
plt.show()
```
