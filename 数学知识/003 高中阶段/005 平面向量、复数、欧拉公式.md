#### 1. 平面向量（高中最强工具，直接把几何变代数）

**高中终极认知（背会高考+奥赛通杀）**
- 向量 = 既有大小又有方向的量  
- 几何问题 → 坐标/向量 → 代数运算 → 答案  
- 数量积（点积）是高考之神

**核心公式（必须手推十遍）**
```
a·b = |a||b|cosθ = x₁x₂ + y₁y₂
|a| = √(x²+y²)
投影公式、方向角、夹角公式全靠它
```

```python
# 向量基本运算现场表演
a = np.array([3, 4])
b = np.array([-1, 5])

print(f"向量 a = {a},  |a| = {np.linalg.norm(a):.0f}")   # 5
print(f"a·b = {np.dot(a,b)}")                          # 17
print(f"夹角 cosθ = {np.dot(a,b)/(5*np.linalg.norm(b)):.3f}")
print(f"θ ≈ {np.rad2deg(np.arccos(np.dot(a,b)/(5*np.linalg.norm(b)))):.1f}°")

# 画图让你彻底明白
class Arrow(FancyArrowPatch):
    def __init__(self, posA, posB, color='red', label=''):
        super().__init__(posA, posB, arrowstyle='->', mutation_scale=20,
                         linewidth=3, color=color, label=label)

fig, ax = plt.subplots(figsize=(10,10))
ax.set_xlim(-2,8); ax.set_ylim(-2,8); ax.grid(alpha=0.3); ax.set_aspect('equal')

ax.add_patch(Arrow((0,0), a, 'blue', 'a'))
ax.add_patch(Arrow((0,0), b, 'red',  'b'))
ax.add_patch(Arrow(a, a+b, 'green', 'b平移'))
ax.add_patch(Arrow((0,0), a+b, 'purple', 'a+b'))

ax.text(3,4.5,'a',fontsize=16,color='blue',weight='bold')
ax.text(-1,5.5,'b',fontsize=16,color='red',weight='bold')
ax.text(2,8,'a+b',fontsize=16,color='purple',weight='bold')

ax.set_title('平面向量加法 + 数量积几何意义（高中向量第一铁律）',fontsize=20)
plt.legend(handles=[Arrow((0,0),(1,1),'blue'), Arrow((0,0),(1,1),'red')], 
           labels=['向量a','向量b'], loc='upper left')
plt.show()
```

#### 2. 复数（高中最“玄幻”的一章，其实最简单）

**高中最强认知**
- 复数 = 实部 + 虚部 = x + yi  
- i² = −1 → 复平面上的“转90°”  
- 复数就是平面上的向量 + 乘法会旋转！

```python
# 复数运算一键搞定
z1 = 3 + 4j
z2 = -1 + 5j

print(f"z1 = {z1}, |z1| = {abs(z1):.0f}")
print(f"z1 × z2 = {z1*z2}")                     # 旋转+伸缩
print(f"|z1×z2| = |z1||z2| = {abs(z1*z2):.0f}")

# 复平面画图
z = np.array([3+4j, -1+5j, (3+4j)*(-1+5j), 1+1j, 2-3j, -4+2j])
plt.figure(figsize=(10,10))
plt.axhline(0,color='k',lw=1); plt.axvline(0,color='k',lw=1)
plt.scatter(z.real, z.imag, c='red', s=100)
for i in range(len(z)):
    plt.text(z[i].real+0.2, z[i].imag, f'z{i+1}', fontsize=14, weight='bold')
plt.grid(alpha=0.3); plt.axis('equal')
plt.xlabel('实部 Re'); plt.ylabel('虚部 Im')
plt.title('复平面：复数就是会旋转的向量！（高中复数核心）',fontsize=20)
plt.show()
```

#### 3. 欧拉公式（全人类最美的公式，高中提前偷跑）

**欧拉公式（背会直接封神）**
e^(iθ) = cos θ + i sin θ  
特别地：e^(iπ) + 1 = 0

```python
# 现场验证欧拉公式（误差10^-16级）
theta = np.linspace(0, 2*np.pi, 500)
euler_left  = np.exp(1j * theta)                 # 复指数
euler_right = np.cos(theta) + 1j*np.sin(theta)   # 欧拉公式

error = np.max(np.abs(euler_left - euler_right))
print(f"欧拉公式误差 = {error:.2e}（本质上为0）")

# 最美的一张图
plt.figure(figsize=(10,10))
plt.plot(euler_left.real, euler_left.imag, 'purple', linewidth=4, label='e^(iθ)')
circle = plt.Circle((0,0),1,color='gray',fill=False,linewidth=2)
plt.gca().add_patch(circle)
plt.axis('equal'); plt.grid(alpha=0.3)
plt.title('欧拉公式：e^(iθ) = cos θ + i sin θ\n全人类最美的公式（高中提前看）',fontsize=20)
plt.xlabel('实部'); plt.ylabel('虚部')
plt.legend(fontsize=16)
plt.show()

# 特别地：e^(iπ) + 1 = 0
print(f"e^(iπ) = {np.exp(1j*np.pi):.20f}")
print(f"e^(iπ) + 1 = {np.exp(1j*np.pi) + 1:.20f}")
```
